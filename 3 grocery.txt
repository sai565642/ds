
#include<stdio.h>
struct Item{
    char name [50];
    float price;
};
int main(){
    int n,i;
    float total=0.0,discount=0.0,finalamount;
    printf("Enter the number of the grocery items:");
    scanf("%d",&n);
    struct Item items[n];
    //input items name and price 
    for (i=0;i<n;i++){
        printf ("\n Enter the name of items%d:",i+1);
        scanf("%s",items[i].name);
        printf(" Enter the price of %s:₹",items[i].name);
        scanf("%f",&items[i].price);
        total+=items[i].price;
    }
    // Display bill summary
    printf("\n ------ Bill Summary-----\n");
    for(i=0;i<n;i++){
        printf("%s:₹%.2f\n",items[i].name,items[i].price);
    }
    printf("\n total bill:₹%.2f",total);
    //Apply Discount
    if(total>1000){
        discount=total*0.10;
        printf("\n discount applied:₹%.2f",discount);
    }
    else{
        printf("\n no discount applied");
    }
    finalamount=total-discount;
    printf("\n final bill amount:₹%.2f\n",finalamount);
    return 0;
    
}

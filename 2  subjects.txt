#include <stdio.h>

struct Student {
    char name[30];
    int usn;
    int marks[5];
    int total_marks;
    float percentage;
};

int main() {
    struct Student s[5]; // Array of 5 students
    int i, j;

    // Input section
    for (i = 0; i < 5; i++) {
        printf("\nEnter details for student %d:\n", i + 1);

        printf("USN: ");
        scanf("%d", &s[i].usn);

        printf("Name: ");
        scanf("%s", s[i].name);

        s[i].total_marks = 0;

        printf("Enter marks for 5 subjects:\n");
        for (j = 0; j < 5; j++) {
            printf("Subject %d: ", j + 1);
            scanf("%d", &s[i].marks[j]);
            s[i].total_marks += s[i].marks[j];
        }

        s[i].percentage = s[i].total_marks / 5.0;
    }

    // Output section
    printf("\n--- Student Details ---\n");
    for (i = 0; i < 5; i++) {
        printf("\nStudent %d:\n", i + 1);
        printf("Name: %s\n", s[i].name);
        printf("USN: %d\n", s[i].usn);
        printf("Total Marks: %d\n", s[i].total_marks);
        printf("Percentage: %.2f%%\n", s[i].percentage);
    }

    return 0;
}
